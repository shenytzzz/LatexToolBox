% Minimal cover page.
\newcommand{\LTBCoverPage}{%
  \clearpage
  \hypersetup{pageanchor=false}%
  \begingroup
  \thispagestyle{empty}%
  \color{LTBText}%
  \vspace*{1.0cm}%
  \noindent{\color{LTBBlue}\rule{\textwidth}{1.15pt}}\par
  \vspace{0.45cm}%
  {\large\bfseries\color{LTBMutedBlue}\CoverTopText\par}%
  \vfill
  \begin{center}
    {\fontsize{34}{40}\selectfont\bfseries\color{LTBBlue}\CoverTitle\par}%
    \vspace{0.55cm}%
    {\Large\color{LTBText}\CoverSubtitle\par}%
    \vspace{1.05cm}%
    {\normalsize\bfseries\color{LTBMutedBlue}\CoverAuthor\par}%
  \end{center}
  \vfill
  \noindent{\color{LTBRule}\rule{\textwidth}{0.55pt}}\par
  \vspace{0.35cm}%
  \noindent{\small\color{LTBMutedBlue}\CoverBottomLeft\hfill\CoverDate\par}%
  \clearpage
  \endgroup
  \hypersetup{pageanchor=true}%
}
